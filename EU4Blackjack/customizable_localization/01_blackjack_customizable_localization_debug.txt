# Win Status
defined_text = {
    name = get_p1w
	text = {
		localisation_key = debug_b_lose
        trigger = {
            event_target:blackjack_player_1_tag = {
                NOT = {
                    check_variable = {
                        which = blackjack_player_1_win_status
                        value = 1
                    }
                }
            }
        }
    }
    
    text = {
        localisation_key = debug_b_push
        trigger = {
            event_target:blackjack_player_1_tag = {

                    NOT = {
                        check_variable = {
                            which = blackjack_player_1_win_status
                            value = 2
                        }
                    }

            }
        }
    }

                        value = 2
                    }
                    NOT = {
                        check_variable = {
                            which = blackjack_player_1_win_status
                            value = 3
                        }
                    }
                }
            }
        }
    }

    text = {
